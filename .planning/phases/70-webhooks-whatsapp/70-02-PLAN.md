```typescript
/**
 * Scans the tenant's audit rows (bounded) and returns the most recent row
 * whose `recipient_provider_id` matches. Best-effort lookup used by webhook
 * handlers (new_relation, message_received). Returns null on no match,
 * malformed JSON, scan failure, or empty store.
 *
 * D-77 (CONTEXT): new_relation arrives up to 8h after the original send,
 * so the scan window must be wide. Default limit 200 covers ~7 days of
 * activity at the 25/day connect cap; tune via options.limit.
 */
export async function findAuditByProviderId(
  providerId: string,
  options?: { limit?: number }
): Promise<AuditRow | null> {
  if (!providerId) return null;
  const limit = options?.limit ?? 200;
  try {
    const kv = getContextKVStore();
    const keys = await kv.list("unipile:audit:");
    let best: AuditRow | null = null;
    let scanned = 0;
    for (const key of keys) {
      if (key.includes(":hash:")) continue; // skip pointer keys
      if (scanned++ >= limit) break;
      const raw = await kv.get(key);
      if (!raw) continue;
      try {
        const row = JSON.parse(raw) as AuditRow;
        if (row.recipient_provider_id === providerId) {
          if (!best || row.timestamp > best.timestamp) best = row;
        }
      } catch { continue; }
    }
    return best;
  } catch {
    return null;
  }
}
```

import { resolveTenantFromAccountId } from "../dispatcher";
import { runWithTenant } from "@/core/request-context";

const log = getLogger("CONNECTOR:unipile-webhook");

export async function handleAccountStatus(payload: Record<string, unknown>): Promise<void> {
  const accountId = String(payload.account_id ?? "");
  const status = String(payload.account_status ?? "");
  if (!accountId || !status) {
    log.warn("[CONNECTOR:unipile-webhook] account_status missing account_id or status", { keys: Object.keys(payload) });
    return;
  }

  const tenantId = await resolveTenantFromAccountId(accountId);
  if (!tenantId) {
    log.warn("[CONNECTOR:unipile-webhook] account_status — no tenant mapping", { accountId, status });
    return;
  }

  await runWithTenant(tenantId, async () => {
    if (isHaltStatus(status)) {
      await writeHaltFlag(accountId, {
        reason: String(payload.account_status_specifics ?? status),
        halted_at: new Date().toISOString(),
        status,
      });
      log.warn("[CONNECTOR:unipile-webhook] account halted", { accountId, status, tenantId });
    } else if (isRecoveryStatus(status)) {
      await clearHaltFlag(accountId);
      log.info("[CONNECTOR:unipile-webhook] account recovered — halt cleared", { accountId, status, tenantId });
    } else {
      log.debug("[CONNECTOR:unipile-webhook] account_status no-op", { accountId, status });
    }
  });
}

import { runWithTenant } from "@/core/request-context";

  writeAuditRow,

  type AuditRow,
} from "../../lib/audit";

const log = getLogger("CONNECTOR:unipile-webhook");

export async function handleNewRelation(payload: Record<string, unknown>): Promise<void> {
  const accountId = String(payload.account_id ?? "");
  const userProviderId = String(payload.user_provider_id ?? "");
  if (!accountId || !userProviderId) {
    log.warn("[CONNECTOR:unipile-webhook] new_relation missing fields", { keys: Object.keys(payload) });
    return;
  }

  const tenantId = await resolveTenantFromAccountId(accountId);
  if (!tenantId) {
    log.warn("[CONNECTOR:unipile-webhook] new_relation — no tenant mapping", { accountId });
    return;
  }

  await runWithTenant(tenantId, async () => {
    const matching = await findAuditByProviderId(userProviderId);
    const acceptedAt = new Date().toISOString();
    if (matching) {
      // D-61: enrich existing audit row in-place (re-write under same audit_id)
      const enriched: AuditRow = { ...matching, accepted_at: acceptedAt };
      await writeAuditRow(enriched);
      log.info("[CONNECTOR:unipile-webhook] new_relation — enriched audit row", {
        audit_id: matching.audit_id, accepted_at: acceptedAt, accountId,
      });
    } else {
      // D-61 fallback: standalone inbound row (D-78 inbound_accept_unknown_origin)
      const row: AuditRow = {
        audit_id: generateAuditId(),
        actor_user_id: "system",
        tool: "webhook:new_relation",
        account_id: accountId,
        params_hash: "inbound",
        result: "inbound_accept_unknown_origin",
        verified: true,
        dedup_hit: false,
        timestamp: acceptedAt,
        recipient_provider_id: userProviderId,
        accepted_at: acceptedAt,
      };
      await writeAuditRow(row);
      log.info("[CONNECTOR:unipile-webhook] new_relation — standalone inbound row", {
        audit_id: row.audit_id, accountId, recipient: userProviderId,
      });
    }
  });
}

  }

  const accountId = String(payload.account_id ?? "");
  const messageId = String(payload.message_id ?? "");
  if (!accountId || !messageId) {
    log.warn("[CONNECTOR:unipile-webhook] message_received missing account_id or message_id", { keys: Object.keys(payload) });
    return;
  }

  const tenantId = await resolveTenantFromAccountId(accountId);
  if (!tenantId) {
    log.warn("[CONNECTOR:unipile-webhook] message_received — no tenant mapping", { accountId });
    return;
  }

  await runWithTenant(tenantId, async () => {
    const senderProviderId = String(payload.attendee_provider_id ?? payload.sender_attendee_id ?? "");
    const lastRepliedAt = new Date().toISOString();
    const contentHash = hashBody(payload);  // D-64 — only the hash leaves the lambda

    const matching = senderProviderId ? await findAuditByProviderId(senderProviderId) : null;
    if (matching) {
      const enriched: AuditRow = { ...matching, last_replied_at: lastRepliedAt };
      await writeAuditRow(enriched);
      log.info("[CONNECTOR:unipile-webhook] message_received — enriched audit row", {
        audit_id: matching.audit_id, last_replied_at: lastRepliedAt, accountId,
      });
    } else {
      const row: AuditRow = {
        audit_id: generateAuditId(),
        actor_user_id: "system",
        tool: "webhook:message_received",
        account_id: accountId,
        params_hash: `inbound:${contentHash}`,  // hash-only — D-64
        result: "inbound_message_unknown_origin",
        verified: true,
        dedup_hit: false,
        timestamp: lastRepliedAt,
        recipient_provider_id: senderProviderId || undefined,
        last_replied_at: lastRepliedAt,
      };
      await writeAuditRow(row);
      log.info("[CONNECTOR:unipile-webhook] message_received — standalone inbound row", {
        audit_id: row.audit_id, accountId, content_hash: contentHash,
      });
    }
  });
}
