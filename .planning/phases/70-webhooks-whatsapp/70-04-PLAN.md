```typescript
/**
 * Phase 70 / Plan 04 — Shared halt-check helper for all 5 destructive
 * Unipile write tools (LinkedIn × 4 + WhatsApp × 1).
 *
 * D-65/D-66 (LOCKED): halt-check is Step 0 of every write tool handler —
 * BEFORE D-49 dedup-first ordering inherited from phase 69. When the
 * account_status webhook handler (Plan 02) sets the halt flag, all
 * subsequent write attempts must short-circuit and write ONE audit row
 * with result="error_account_halted". No dedup, no rate-limit, no SDK.
 *
 * Anti-drift rationale: 5 sites would otherwise carry near-identical
 * 15-LOC blocks. The cost of one tiny helper function << drift risk.
 * Mirrors the phase 69 Plan 01 account.ts extraction pattern.
 *
 * Failure-mode (KV read fails): returns {halted: false}. This is
 * fail-OPEN for the gate — the gate is defense-in-depth on top of
 * Unipile's own account-status enforcement; the rate-limiter (which IS
 * security-critical and fails CLOSED) is still upstream of the SDK call.
 * Logging the KV failure surfaces operator action.
 */

import { readHaltFlag } from "../webhook/halt-flag";
import { writeAuditRow, generateAuditId, type AuditResult } from "./audit";
import { getLogger } from "@/core/logging";
import { toMsg } from "@/core/error-utils";

const log = getLogger("CONNECTOR:unipile");

export interface HaltCheckArgs {
  account_id: string | undefined | null;
  tool: string;
  actor_user_id: string;
  audit_id?: string;  // optional — caller may pre-generate
}

export interface HaltCheckClear { halted: false }
export interface HaltCheckBlocked {
  halted: true;
  audit_id: string;
  envelope_fields: {
    provider_ok: false;
    verified: false;
    crm_sync: "pending";
    dedup_hit: false;
    audit_id: string;
    error: "error_account_halted";
    halt_reason: string;
    halted_at: string;
  };
}
export type HaltCheckResult = HaltCheckClear | HaltCheckBlocked;

export async function haltCheck(args: HaltCheckArgs): Promise<HaltCheckResult> {
  if (!args.account_id) return { halted: false };
  let flag;
  try {
    flag = await readHaltFlag(args.account_id);
  } catch (err) {
    log.warn("[CONNECTOR:unipile] halt-check KV read failed; fail-OPEN", { err: toMsg(err), account_id: args.account_id, tool: args.tool });
    return { halted: false };
  }
  if (!flag) return { halted: false };

  const auditId = args.audit_id ?? generateAuditId();
  await writeAuditRow({
    audit_id: auditId,
    actor_user_id: args.actor_user_id,
    tool: args.tool,
    account_id: args.account_id,
    params_hash: "halted",
    result: "error_account_halted" as AuditResult,
    verified: false,
    dedup_hit: false,
    timestamp: new Date().toISOString(),
  });

  return {
    halted: true,
    audit_id: auditId,
    envelope_fields: {
      provider_ok: false,
      verified: false,
      crm_sync: "pending",
      dedup_hit: false,
      audit_id: auditId,
      error: "error_account_halted",
      halt_reason: flag.reason,
      halted_at: flag.halted_at,
    },
  };
}
```
