**dispatcher.ts** — Ship from PATTERNS.md §dispatcher.ts + tenant-routing helper. Key implementation notes:

```typescript
import { getLogger } from "@/core/logging";
import { getAccountTenant, writeAccountTenantMapping } from "./account-tenant-index";

const log = getLogger("CONNECTOR:unipile-webhook");

// Hook table — Plan 02 sets these. Default = log.warn no-op.
type EventHandler = (payload: Record<string, unknown>) => Promise<void>;
const noop: EventHandler = async (p) => {
  log.warn("[CONNECTOR:unipile-webhook] handler not registered", { event: p.event ?? p.account_status });
};
export const _handlers = {
  messageReceived: noop as EventHandler,
  newRelation: noop as EventHandler,
  accountStatus: noop as EventHandler,
};

export function getIdempotencyKey(p: Record<string, unknown>): string | null {
  const event = String(p.event ?? "");
  if (event === "message_received" && typeof p.message_id === "string") return p.message_id;
  if (event === "new_relation" && typeof p.account_id === "string" && typeof p.user_provider_id === "string") {
    return `${p.account_id}:${p.user_provider_id}`;
  }
  if (typeof p.account_id === "string" && typeof p.account_status === "string") {
    return `${p.account_id}:${p.account_status}:${p.timestamp ?? Date.now()}`;
  }
  return null;
}

export async function dispatchEventAsync(payload: Record<string, unknown>): Promise<void> {
  const event = String(payload.event ?? "");
  if (event === "message_received" && payload.is_sender === true) {
    log.debug("[CONNECTOR:unipile-webhook] skip outbound echo", { message_id: payload.message_id });
    return;
  }
  switch (event) {
    case "message_received": return _handlers.messageReceived(payload);
    case "new_relation":     return _handlers.newRelation(payload);
    default:
      if (typeof payload.account_status === "string") return _handlers.accountStatus(payload);
      log.warn("[CONNECTOR:unipile-webhook] unknown event type", { event, keys: Object.keys(payload) });
  }
}

export async function resolveTenantFromAccountId(accountId: string): Promise<string | null> {
  return getAccountTenant(accountId);
}

export async function claimAccountForTenant(accountId: string, tenantId: string): Promise<void> {
  await writeAccountTenantMapping(accountId, tenantId);
}
```

Rationale for the hook table: Plan 02 must wire handlers, but Plan 01 ships independently (and the dispatcher must compile + be unit-testable). The hook indirection avoids cyclic imports and lets Plan 02 do `import { _handlers } from "../dispatcher"; _handlers.messageReceived = handleMessageReceived;` from a handlers/index.ts barrel that the route imports for side-effect registration.

```typescript
import { getKVStore } from "@/core/kv-store";  // ROOT scope per D-56 + Pitfall — needs allowlist entry (Task 3)
import { composeRequestPipeline, rehydrateStep, bodyParseStep, type PipelineContext } from "@/core/pipeline";
import { getConfig } from "@/core/config-facade";
import { getLogger } from "@/core/logging";
import { toMsg } from "@/core/error-utils";
import { verifyUnipileWebhook } from "@/connectors/unipile/webhook/verifier";
import { dispatchEventAsync, getIdempotencyKey } from "@/connectors/unipile/webhook/dispatcher";
import "@/connectors/unipile/webhook/handlers"; // SIDE-EFFECT: registers handlers (Plan 02 — file is created later but tolerant)

const MAX_PAYLOAD_BYTES = 256 * 1024; // 256KB — Unipile bodies are 1-2KB
const log = getLogger("CONNECTOR:unipile-webhook");

async function unipileWebhookHandler(ctx: PipelineContext): Promise<Response> {
  const secret = getConfig("UNIPILE_WEBHOOK_SECRET");
  if (!secret) {
    return Response.json({ error: "webhook_not_configured" }, { status: 503 });
  }

  // D-77: bodyParseStep already JSON-parsed; re-serialize for HMAC over raw body
  const parsed = ctx.parsedBody;
  const rawBody: string = typeof parsed === "string" ? parsed : JSON.stringify(parsed ?? "");

  // Signature verification (D-52 dual mode)
  const result = verifyUnipileWebhook(rawBody, ctx.request.headers, secret);
  log.info("[CONNECTOR:unipile-webhook] auth attempt", { mode: result.mode, ok: result.ok });
  if (!result.ok) {
    return Response.json({ error: "invalid_signature", reason: result.reason }, { status: 401 });
  }

  // Payload sanity — must be object after bodyParseStep
  const payload = (typeof parsed === "object" && parsed !== null) ? parsed as Record<string, unknown> : null;
  if (!payload) {
    return Response.json({ error: "invalid_payload" }, { status: 400 });
  }

  // D-54 idempotency
  const idemKey = getIdempotencyKey(payload);
  if (idemKey) {
    const kv = getKVStore();
    const setRes = await kv.setIfNotExists?.(`unipile:webhook:event:${idemKey}`, "1", { ttlSeconds: 24 * 3600 });
    if (setRes && !setRes.ok) {
      log.info("[CONNECTOR:unipile-webhook] duplicate event, skipping dispatch", { idemKey });
      return Response.json({ ok: true, deduped: true }, { status: 200 });
    }
  }

  // D-55 fire-and-forget — return 200 fast, run handlers async
  void dispatchEventAsync(payload).catch((err) =>
    log.error("[CONNECTOR:unipile-webhook] dispatch failed", { err: toMsg(err), event: payload.event ?? payload.account_status })
  );
  return Response.json({ ok: true }, { status: 200 });
}

export const POST = composeRequestPipeline(
  [rehydrateStep, bodyParseStep({ maxBytes: MAX_PAYLOAD_BYTES })],
  unipileWebhookHandler
);
```

```typescript
#!/usr/bin/env tsx
import "dotenv/config";
import { UnipileClient } from "unipile-node-sdk";

const DSN = process.env.UNIPILE_DSN;
const TOKEN = process.env.UNIPILE_TOKEN;
const SECRET = process.env.UNIPILE_WEBHOOK_SECRET;
const PUBLIC_URL = process.env.KEBAB_PUBLIC_URL;

if (!DSN || !TOKEN || !SECRET || !PUBLIC_URL) {
  console.error("Missing UNIPILE_DSN / UNIPILE_TOKEN / UNIPILE_WEBHOOK_SECRET / KEBAB_PUBLIC_URL");
  process.exit(1);
}

const client = new UnipileClient(`https://${DSN}`, TOKEN);
const url = `${PUBLIC_URL.replace(/\/$/, "")}/api/unipile/webhook`;

interface ExistingWebhook { id: string; source?: string; name?: string; request_url?: string }
const existing = (await client.webhook.getAll() as { items?: ExistingWebhook[] }).items ?? [];

const desired = [
  { source: "messaging",      name: "unipile-messaging",      mode: "typed" as const },
  { source: "account_status", name: "unipile-account-status", mode: "typed" as const },
  { source: "users",          name: "unipile-users",          mode: "escape" as const },
];

let created = 0, skipped = 0;
for (const d of desired) {
  const already = existing.find(e => e.source === d.source && e.request_url === url);
  if (already) { console.log(`[skip] ${d.source} already present (id=${already.id})`); skipped++; continue; }
  const body = {
    source: d.source,
    request_url: url,
    name: d.name,
    headers: [{ key: "Unipile-Auth", value: SECRET }],
  };
  if (d.mode === "typed") {
    await client.webhook.create(body as never);
  } else {
    // D-68 escape hatch for source: "users"
    await client.request.send({ path: ["webhooks"], method: "POST", body: body as never });
  }
  console.log(`[create] ${d.source} → ${url}`);
  created++;
}
console.log(`Done: ${created} created, ${skipped} skipped`);
```
