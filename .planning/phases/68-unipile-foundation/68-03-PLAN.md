    import { createHash } from "node:crypto";
    import { getContextKVStore } from "@/core/request-context";
    import { getUnipileClient } from "../client";
    import { withRetry } from "./retry";

    /** D-10: cache TTL is 30 days. Upstash respects EX, Filesystem ignores (dev only). */
    export const URN_TTL_SECONDS = 30 * 24 * 60 * 60; // 2,592,000

    /**
     * Accepts:
     *   - https://linkedin.com/in/<slug>
     *   - https://www.linkedin.com/in/<slug>
     *   - https://<locale>.linkedin.com/in/<slug>   (fr, de, es, it, pt, nl, pl, tr, zh, ja, ko, ar, ru)
     *   - linkedin.com/in/<slug>                    (protocol stripped — we add https://)
     * Per D-12.
     */
    const SLUG_RE =
      /^(?:https?:\/\/)?(?:www\.|(?:fr|de|es|it|pt|nl|pl|tr|zh|ja|ko|ar|ru)\.)?linkedin\.com\/in\/([a-zA-Z0-9\-_%]+)\/?$/;

    /** D-12 normalization: lowercase slug, strip trailing slash, strip locale prefix, force https://linkedin.com/in/<slug>. */
    export function normalizeProfileUrl(input: string): string {
      const trimmed = input.trim();
      const m = SLUG_RE.exec(trimmed);
      if (!m || !m[1]) {
        throw new Error(
          `Invalid LinkedIn profile URL: ${input}. Supported formats: https://linkedin.com/in/<slug>, https://www.linkedin.com/in/<slug>, https://<locale>.linkedin.com/in/<slug>.`
        );
      }
      const slug = m[1].toLowerCase();
      return `https://linkedin.com/in/${slug}`;
    }

    /** SHA-256 of the normalized URL, truncated to 16 hex chars for KV-key brevity. Per Claude's Discretion. */
    function urlHash(normalized: string): string {
      return createHash("sha256").update(normalized).digest("hex").slice(0, 16);
    }

    /** Exported so the admin DELETE route can compute the SAME key. */
    export function urnCacheKey(normalizedUrl: string): string {
      return `unipile:urn:${urlHash(normalizedUrl)}`;
    }

    interface UrnCacheRow {
      urn: string;
      resolved_at: string;
    }

    /**
     * Resolve a LinkedIn profile URL to a Unipile `provider_id` URN, using KV cache (30-day TTL).
     * On Unipile 429 / unrecoverable error, throws — caller surfaces UnipileRateLimitError or similar.
     * Per D-09 / D-10 / D-18.
     */
    export async function resolveProviderId(
      rawUrl: string,
      accountId: string
    ): Promise<{ provider_id: string; from_cache: boolean }> {
      const normalized = normalizeProfileUrl(rawUrl);
      const kv = getContextKVStore();
      const key = urnCacheKey(normalized);

      const cached = await kv.get(key);
      if (cached) {
        try {
          const parsed = JSON.parse(cached) as UrnCacheRow;
          if (parsed && typeof parsed.urn === "string") {
            return { provider_id: parsed.urn, from_cache: true };
          }
        } catch {
          // Corrupt cache row — fall through to fresh resolve.
        }
      }

      // Cache miss → call SDK. SLUG_RE guarantees this strip yields a non-empty identifier.
      const slug = normalized.slice("https://linkedin.com/in/".length);
      const client = getUnipileClient();
      const profile = await withRetry(() =>
        client.users.getProfile({ account_id: accountId, identifier: slug })
      );
      const providerId = (profile as { provider_id?: unknown }).provider_id;
      if (typeof providerId !== "string" || providerId.length === 0) {
        throw new Error(
          `Unipile getProfile did not return a provider_id for ${normalized}; got ${JSON.stringify(profile)}`
        );
      }

      const row: UrnCacheRow = { urn: providerId, resolved_at: new Date().toISOString() };
      await kv.set(key, JSON.stringify(row), URN_TTL_SECONDS);
      return { provider_id: providerId, from_cache: false };
    }
    ```

    import { NextResponse } from "next/server";
    import { withAdminAuth } from "@/core/with-admin-auth";
    import type { PipelineContext } from "@/core/pipeline";
    import { toMsg } from "@/core/error-utils";
    import { getKVStore } from "@/core/kv-store";
    import { normalizeProfileUrl, urnCacheKey } from "@/connectors/unipile/lib/identifiers";

    // KV-ALLOWLIST-EXEMPT: cache eviction for unipile URN keys uses raw getKVStore() (root scope)
    // rather than getContextKVStore() because an admin operator may need to evict poisoned/stale
    // entries across all tenants from a single dashboard action. Mirrors the pattern of
    // app/api/admin/rate-limits/route.ts (Phase 53). See .planning/phases/68-unipile-foundation/
    // 68-CONTEXT.md D-18 for the locked decision.
    //
    // NOTE: tenant-prefixed callers (resolver in src/connectors/unipile/lib/identifiers.ts) write
    // the key as `tenant:<id>:unipile:urn:<hash>`. This root-scope delete wipes ONLY the
    // unscoped `unipile:urn:<hash>` key. To evict a specific tenant's cached URN, that tenant
    // must call the same eviction logic from within their request context (not exposed here in
    // phase 68 — open as a phase 71 enhancement if needed).

    async function deleteHandler(ctx: PipelineContext): Promise<Response> {
      try {
        const url = new URL(ctx.request.url);
        const profileUrl = url.searchParams.get("profile_url");
        if (!profileUrl) {
          return NextResponse.json(
            { ok: false, error: "profile_url query parameter is required" },
            { status: 400 }
          );
        }

        let normalized: string;
        try {
          normalized = normalizeProfileUrl(profileUrl);
        } catch (err) {
          return NextResponse.json({ ok: false, error: toMsg(err) }, { status: 400 });
        }

        const key = urnCacheKey(normalized);
        const kv = getKVStore();
        await kv.delete(key);

        return NextResponse.json({
          ok: true,
          evicted: true,
          key,
          normalized_url: normalized,
        });
      } catch (err) {
        return NextResponse.json({ ok: false, error: toMsg(err) }, { status: 500 });
      }
    }

    export const DELETE = withAdminAuth(deleteHandler);
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| MCP tool / handler → KV (URN cache reads/writes) | Tenant-scoped via getContextKVStore — tenant A cannot read tenant B's URN cache. |
| Admin operator → KV (URN cache eviction) | Root-scope via getKVStore — admin can evict the unscoped key, but tenant-prefixed copies survive until they expire naturally. |
| Public URL input (`profile_url` query) → KV key derivation | createHash sanitizes; normalize rejects unsupported formats. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-68-03-01 | Tampering | URN cache poisoning by malicious tenant | mitigate | Per-tenant KV prefix via getContextKVStore — tenant A cannot poison tenant B's cache. Admin DELETE is the operator-only fix path. |
| T-68-03-02 | Information Disclosure | URN value (provider_id) leaked across tenants | mitigate | Tenant-prefixed keys. Even if two tenants resolve the same public profile URL, each gets its own KV row under `tenant:<id>:unipile:urn:<hash>`. |
| T-68-03-03 | Spoofing | Forged DELETE request without admin cookie | mitigate | `withAdminAuth` HOC verifies the admin cookie via the pipeline's existing auth step (Phase 50 BRAND-02). Failure returns 401. |
| T-68-03-04 | Tampering | Malformed `profile_url` triggers ReDoS via SLUG_RE | mitigate | SLUG_RE has bounded alternation and a simple character class — not vulnerable to catastrophic backtracking. Input trimmed before match. |
| T-68-03-05 | Information Disclosure | Cache eviction logs the full profile_url | accept | `profile_url` is a public LinkedIn slug, not PII (the slug is what appears in the public-facing URL). Logging is acceptable and aids operator debugging. The cached `provider_id` is also not sensitive. |
| T-68-03-06 | Denial of Service | Operator bulk-evicts the cache | accept | Cost of bulk eviction is ~1 KV.delete per call. At Cadens scale, an operator scripting 1000 evictions is fine. The next resolve fills the cache again. |
</threat_model>
